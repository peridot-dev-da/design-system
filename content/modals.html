Modal Components

Interactive modal dialogs with multiple sizes, proper accessibility features, and theme support.

Basic Modal

A simple modal with header, body, and footer sections. 

<!-- Trigger Button -->
<button class="btn btn-primary" data-modal-target="basicModal">
    Open Basic Modal
</button>

<!-- Modal -->
<div class="modal" id="basicModal">
    <div class="modal-dialog">
        <div class="modal-content">
            <div class="modal-header">
                <h3 class="modal-title">
                    <i data-feather="info" class="icon"></i>
                    Basic Modal
                </h3>
                <button type="button" class="modal-close" data-modal-dismiss>
                    <i data-feather="x"></i>
                </button>
            </div>
            <div class="modal-body">
                <p>This is a basic modal with standard styling.</p>
            </div>
            <div class="modal-footer">
                <button type="button" class="btn btn-secondary" data-modal-dismiss>
                    Cancel
                </button>
                <button type="button" class="btn btn-primary">
                    Save Changes
                </button>
            </div>
        </div>
    </div>
</div>

Static Backdrop

A modal that cannot be closed by clicking the backdrop. 

<!-- Trigger with static backdrop -->
<button class="btn btn-warning" 
        data-modal-target="staticModal" 
        data-modal-backdrop="static">
    Static Modal
</button>

<!-- Or via JavaScript -->
<script>
showModal('staticModal', { backdrop: 'static' });
</script>

JavaScript API

Control modals programmatically with the JavaScript API. 

// Basic modal operations
showModal('modalId');                    // Show modal
hideModal('modalId');                    // Hide modal
toggleModal('modalId');                  // Toggle modal

// With options
showModal('modalId', {
    backdrop: 'static',                  // true, false, or 'static'
    keyboard: false,                     // Disable ESC key
    focus: true                          // Auto-focus modal
});

// Get modal instance for advanced operations
const modal = PDSModals.getInstance('modalId');
modal.setTitle('New Title');
modal.setBody('<p>New content</p>');
modal.setFooter('<button class="btn btn-primary">OK</button>');

// Event listeners
document.getElementById('modalId').addEventListener('modal.shown', (e) => {
    console.log('Modal shown:', e.detail.modal);
});

document.getElementById('modalId').addEventListener('modal.hidden', (e) => {
    console.log('Modal hidden:', e.detail.modal);
});
